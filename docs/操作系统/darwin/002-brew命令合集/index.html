brew安装

/usr/bin/ruby -e "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/master/install)"

常用示例

# 更新 Homebrew 自己
brew update

# 查看哪些安装包需要更新
brew outdated

# 更新所有的包
brew upgrade 

# 更新指定的包
brew upgrade $FORMULA

# 清理所有包的旧版本
brew cleanup

# 清理指定包的旧版本
brew cleanup $FORMULA

# 查看可清理的旧版本包，不执行实际操作    
brew cleanup -n          

# 锁定某个包不更新
brew pin $FORMULA

# 取消锁定
brew unpin $FORMULA

# 显示安装了包数量，文件数量，和总占用空间
brew info

# 显示某个包的信息
brew info $FORMULA

# 查看已安装的包的依赖，树形显示
brew deps --installed --tree 

# 列出已安装包
brew list

# 删除某个包
brew rm $FORMULA

# 删除所有版本
brew uninstall --force $FORMULA
